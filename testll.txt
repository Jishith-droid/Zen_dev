fn split(string s, string delim) List<string> {

  List<string> result = []

  int n = length(s)
  int d = length(delim)
  
  if (n == 0) {
    return [""]
  }

  if (d == 0) {
    return [s]
  }

  if (d > n) {
    return [s]
  }

  string current = ""
  int i = 0

  while (i < n) {

    // =========================
    // MATCH CHECK (NO TEMP STRING)
    // =========================
    int match = 1
    int j = 0

    while (j < d && (i + j) < n) {

      if (charAt(s, i + j) != charAt(delim, j)) {
        match = 0
        break
      }

      j += 1
    }

    // full match found
    if (match == 1 && j == d) {

      result.push(current)
      current = ""

      i += d
      continue
    }

    // normal accumulation
    current = current + charAt(s, i)
    i += 1
  }

  result.push(current)
  
  return result
}

fn join(List<string> items, string sep) string {

  int n = items.length

  if (n == 0) {
    return ""
  }

  string result = ""
  int i = 0

  while (i < n) {

    result += items[i]

    if (i < n - 1) {
      result += sep
    }

    i += 1
  }

  return result
}
